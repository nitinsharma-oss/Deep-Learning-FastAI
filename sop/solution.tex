\section*{Setup and Notation}
Both links $L\in\{B,E\}$ (Bob, Eve) have instantaneous SNR $\gamma_L$ with the three-region PDF stated in the problem, parameters $(K_L,\gamma_{oL},p_L,m_{1L},m_{2L},M_{1L},M_{2L})$, and support breakpoints
\[
a_L=\gamma_{oL}m_{1L}m_{2L},\quad b_L=\gamma_{oL}m_{2L}M_{1L},\quad
c_L=\gamma_{oL}m_{1L}M_{2L},\quad d_L=\gamma_{oL}M_{1L}M_{2L},
\]
with the ordering $a_L<b_L\le c_L<d_L$ (equivalently $M_{2L}/m_{2L}\ge M_{1L}/m_{1L}$) and $0<m_{1L}<M_{1L}\le 1$ so that all incomplete-Beta arguments lie in $[0,1]$. Write
\[
B_i^L(x):=\Bx{x}{p_L+i},\quad i=0,1,2,\qquad
\Delta B_i^L:=B_i^L(M_{1L})-B_i^L(m_{1L}).
\]
Three antiderivative identities, each verified by direct differentiation using $\frac{d}{dx}\Bx{x}{a}=x^{a-1}(1-x)^{-1/2}$:
\begin{align}
\int B_0(x)\,dx &= G(x):= x\,B_0(x)-B_1(x), \tag{P1}\\
\int x\,B_0(x)\,dx &= H(x):= \tfrac{x^2}{2}B_0(x)-\tfrac12 B_2(x), \tag{P2}\\
\int G(x)\,dx &= \Phi(x):= \tfrac{x^2}{2}B_0(x)-x\,B_1(x)+\tfrac12 B_2(x). \tag{P3}
\end{align}
Subscripts $G_B,H_E,\dots$ indicate which link's $p$ is used.

\section*{Task 1: Threshold at $R_s=0.5$}
$2^{2R_s}=2$ and $\tfrac{3\pi e}{6}(2^{2R_s}-1)=\tfrac{\pi e}{2}$, hence
\[
\boxed{\ \tau(\gamma_E)=2\gamma_E+\frac{\pi e}{2}\ },\qquad \frac{\pi e}{2}\approx 4.269867,
\]
strictly increasing in $\gamma_E$, as required for the boundary split.

\section*{Task 2: The Two PDFs and the Normalization Constant}
For $L\in\{B,E\}$,
\[
f_{\gamma_L}(\gamma)=
\begin{cases}
K_L\!\left[B_0^L\!\big(\tfrac{\gamma}{\gamma_{oL}m_{2L}}\big)-B_0^L(m_{1L})\right], & a_L\le\gamma<b_L,\\[2mm]
K_L\!\left[B_0^L(M_{1L})-B_0^L(m_{1L})\right], & b_L\le\gamma<c_L,\\[2mm]
K_L\!\left[B_0^L(M_{1L})-B_0^L\!\big(\tfrac{\gamma}{\gamma_{oL}M_{2L}}\big)\right], & c_L\le\gamma\le d_L,\\[2mm]
0,&\text{otherwise.}
\end{cases}
\]
\textbf{Normalization.} Integrating region by region with (P1) (substituting $u=\gamma/(\gamma_{oL}m_{2L})$ in Region~1 and $u=\gamma/(\gamma_{oL}M_{2L})$ in Region~3) gives the three contributions
\[
K_L\gamma_{oL}m_{2L}\big[M_{1L}\Delta B_0^L-\Delta B_1^L\big],\qquad
K_L\gamma_{oL}(m_{1L}M_{2L}-m_{2L}M_{1L})\Delta B_0^L,\qquad
K_L\gamma_{oL}M_{2L}\big[\Delta B_1^L-m_{1L}\Delta B_0^L\big].
\]
All $\Delta B_0^L$ terms cancel and the sum telescopes to $K_L\gamma_{oL}(M_{2L}-m_{2L})\Delta B_1^L$, so
\[
\boxed{\ K_L=\frac{1}{\gamma_{oL}\,(M_{2L}-m_{2L})\,\big[B\big(M_{1L};p_L{+}1,\tfrac12\big)-B\big(m_{1L};p_L{+}1,\tfrac12\big)\big]}\ }
\]

\section*{Task 3: Piecewise CDF of $\gamma_B$}
Direct integration of each PDF region using (P1):
\begin{align*}
\textbf{Region 1 } (a_B\le x<b_B):\quad
&F_{\gamma_B}(x)=K_B\Big\{\gamma_{oB}m_{2B}\Big[G_B\!\big(\tfrac{x}{\gamma_{oB}m_{2B}}\big)-G_B(m_{1B})\Big]-B_0^B(m_{1B})\,(x-a_B)\Big\},\\[1mm]
\textbf{Region 2 } (b_B\le x<c_B):\quad
&F_{\gamma_B}(x)=F_{\gamma_B}(b_B)+K_B\,\Delta B_0^B\,(x-b_B),\\[1mm]
\textbf{Region 3 } (c_B\le x\le d_B):\quad
&F_{\gamma_B}(x)=F_{\gamma_B}(c_B)+K_B\Big\{B_0^B(M_{1B})(x-c_B)-\gamma_{oB}M_{2B}\Big[G_B\!\big(\tfrac{x}{\gamma_{oB}M_{2B}}\big)-G_B(m_{1B})\Big]\Big\},
\end{align*}
with the continuity constants
\[
F_{\gamma_B}(b_B)=K_B\,\gamma_{oB}m_{2B}\big[M_{1B}\Delta B_0^B-\Delta B_1^B\big],\qquad
F_{\gamma_B}(c_B)=F_{\gamma_B}(b_B)+K_B\,\Delta B_0^B\,\gamma_{oB}\big(m_{1B}M_{2B}-m_{2B}M_{1B}\big),
\]
and $F_{\gamma_B}(x)=0$ for $x<a_B$, $F_{\gamma_B}(x)=1$ for $x\ge d_B$.
\emph{Consistency:} evaluating Region~3 at $x=d_B$ and summing all contributions gives exactly $K_B\gamma_{oB}(M_{2B}-m_{2B})\Delta B_1^B=1$. \checkmark

\textbf{Affine compact form (used in Task 5).} Each region reads
$F_{\gamma_B}(x)=c_r+\lambda_r x+\mu_r\,G_B\!\big(x/w_r\big)$, $r=1,2,3$, with
\[
\begin{array}{lll}
w_1=\gamma_{oB}m_{2B}, & \mu_1=K_B w_1, & \lambda_1=-K_B B_0^B(m_{1B}),\\
 & \mu_2=0, & \lambda_2=K_B\,\Delta B_0^B,\\
w_3=\gamma_{oB}M_{2B}, & \mu_3=-K_B w_3, & \lambda_3=K_B B_0^B(M_{1B}),
\end{array}
\]
and $c_r$ fixed by continuity at $a_B,b_B,c_B$. (The CDF of $\gamma_E$ is identical with $E$ subscripts.)

\section*{Task 4: Critical Value $\gamma_E^{*}$}
Solving $2\gamma_E^{*}+\tfrac{\pi e}{2}=d_B=\gamma_{oB}M_{1B}M_{2B}$:
\[
\boxed{\ \gamma_E^{*}=\frac{\gamma_{oB}M_{1B}M_{2B}}{2}-\frac{\pi e}{4}\ }
\]
If $d_B\le \tfrac{\pi e}{2}$ then $\tau(0)\ge d_B$ and $\mathrm{SOP}=1$ trivially.

\section*{Task 5: Splitting and Evaluating the SOP Integral}
\[
\mathrm{SOP}=\underbrace{\int_{0}^{\gamma_E^{*}}F_{\gamma_B}\big(2\gamma+\tfrac{\pi e}{2}\big)f_{\gamma_E}(\gamma)\,d\gamma}_{\mathcal I}
\;+\;\underbrace{1-F_{\gamma_E}(\gamma_E^{*})}_{\mathcal S}.
\]
\textbf{Saturation term $\mathcal S$ (fully closed form).} Evaluate $F_{\gamma_E}(\gamma_E^*)$ with the Task-3 CDF ($E$ subscripts) in whichever region contains $\gamma_E^*$; $\mathcal S=0$ if $\gamma_E^*\ge d_E$, and $\mathrm{SOP}=1$ if $\gamma_E^*\le a_E$.

\textbf{Integral term $\mathcal I$.} The integrand vanishes unless $f_{\gamma_E}>0$ and $\tau(\gamma)\ge a_B$. With $t_x:=\tfrac{x-\pi e/2}{2}$,
\[
\mathcal I=\int_{\mathcal L}^{\mathcal U}F_{\gamma_B}(2\gamma+\tfrac{\pi e}{2})f_{\gamma_E}(\gamma)\,d\gamma,\qquad
\mathcal L=\max\{a_E,\,t_{a_B},\,0\},\quad \mathcal U=\min\{d_E,\,\gamma_E^*\},
\]
zero if $\mathcal U\le\mathcal L$. Partition $[\mathcal L,\mathcal U]$ at whichever of $\{t_{b_B},t_{c_B},b_E,c_E\}$ fall inside (at most $N\le5$ subintervals $[u_j,v_j]$). On each, Bob's CDF is in a fixed region $r_j$ and Eve's PDF in a fixed region $s_j$, so with Eve constants
\[
q_1=-K_E B_0^E(m_{1E}),\ \nu_1=K_E,\ u_1=\gamma_{oE}m_{2E};\quad
q_2=K_E\Delta B_0^E,\ \nu_2=0;\quad
q_3=K_E B_0^E(M_{1E}),\ \nu_3=-K_E,\ u_3=\gamma_{oE}M_{2E},
\]
and $\tilde A_r=c_r+\tfrac{\pi e}{2}\lambda_r$, $\tilde B_r=2\lambda_r$, $\alpha_r=2/w_r$, $\delta_r=\tfrac{\pi e}{2w_r}$, $\sigma_s=1/u_s$, the integrand is
\[
\Big[\tilde A_r+\tilde B_r\gamma+\mu_r G_B\big(\alpha_r\gamma+\delta_r\big)\Big]
\Big[q_s+\nu_s B_0^E(\sigma_s\gamma)\Big].
\]
Expanding yields exactly four term types on $[u,v]$:
\begin{align*}
\text{(T1)}\quad & q_s\Big[\tilde A_r(v-u)+\tfrac{\tilde B_r}{2}(v^2-u^2)\Big] && \text{elementary},\\
\text{(T2)}\quad & \nu_s\Big[\tfrac{\tilde A_r}{\sigma_s}G_E(\sigma_s\gamma)+\tfrac{\tilde B_r}{\sigma_s^2}H_E(\sigma_s\gamma)\Big]_u^v && \text{closed form via (P1),(P2)},\\
\text{(T3)}\quad & \frac{\mu_r q_s}{\alpha_r}\Big[\Phi_B(\alpha_r\gamma+\delta_r)\Big]_u^v && \text{closed form via (P3)},\\
\text{(T4)}\quad & \mu_r\nu_s\Big[\alpha_r J_{1,0}+\delta_r J_{0,0}-J_{0,1}\Big] && \text{cross term (below)},
\end{align*}
where the canonical cross integral is
\[
J_{n,i}(u,v):=\int_u^v \gamma^{\,n}\,\Bx{\alpha_r\gamma+\delta_r}{p_B{+}i}\;\Bx{\sigma_s\gamma}{p_E}\,d\gamma,\qquad n\in\{0,1\},\ i\in\{0,1\}.
\]
\textbf{Exact reduction of $J_{n,i}$.} Using $\Bx{x}{p}=\sum_{k\ge0}\frac{(1/2)_k}{k!\,(p+k)}x^{p+k}$ (absolutely convergent on $[0,1]$, which covers the entire support),
\[
J_{n,i}=\sum_{k=0}^{\infty}\frac{(1/2)_k\,\sigma_s^{p_E+k}}{k!\,(p_E+k)}\int_u^v \gamma^{\,n+p_E+k}\,\Bx{\alpha_r\gamma+\delta_r}{p_B{+}i}\,d\gamma,
\]
and each inner integral is, after one integration by parts, a boundary term plus an Euler-type integral with three linear factors, hence expressible exactly in Appell $F_1$ hypergeometric functions. If $p_B\in\mathbb Z^+$ (or $p_E\in\mathbb Z^+$), $J_{n,i}$ collapses to a \emph{finite elementary} expression in powers and $\sqrt{1-x}$ terms. For general real $p_B,p_E$ no incomplete-Beta/elementary form exists; $J_{n,i}$ is a smooth finite-interval integral evaluable to arbitrary precision. Note $\mu_{r}\nu_{s}=0$ whenever either link sits in its flat Region~2, so at most a few $J$'s ever appear.

\section*{Task 6: Final SOP Expression}
\[
\boxed{
\begin{aligned}
\mathrm{SOP}=\;&\sum_{j=1}^{N}\Bigg\{
q_{s_j}\Big[\tilde A_{r_j}(v_j-u_j)+\tfrac{\tilde B_{r_j}}{2}(v_j^2-u_j^2)\Big]
+\nu_{s_j}\Big[\tfrac{\tilde A_{r_j}}{\sigma_{s_j}}G_E(\sigma_{s_j}\gamma)+\tfrac{\tilde B_{r_j}}{\sigma_{s_j}^2}H_E(\sigma_{s_j}\gamma)\Big]_{u_j}^{v_j}\\
&\qquad+\frac{\mu_{r_j}q_{s_j}}{\alpha_{r_j}}\Big[\Phi_B(\alpha_{r_j}\gamma+\delta_{r_j})\Big]_{u_j}^{v_j}
+\mu_{r_j}\nu_{s_j}\Big[\alpha_{r_j}J_{1,0}^{(j)}+\delta_{r_j}J_{0,0}^{(j)}-J_{0,1}^{(j)}\Big]\Bigg\}
\;+\;1-F_{\gamma_E}(\gamma_E^{*}),
\end{aligned}}
\]
with $\gamma_E^{*}=\tfrac{\gamma_{oB}M_{1B}M_{2B}}{2}-\tfrac{\pi e}{4}$ and all constants defined above in closed form.
\emph{Limits:} $\gamma_E^*\ge d_E\Rightarrow$ saturation term $=0$, $\mathcal U=d_E$; $\gamma_E^*\le a_E\Rightarrow \mathrm{SOP}=1$; $t_{a_B}\ge d_E\Rightarrow \mathrm{SOP}=1-F_{\gamma_E}(\gamma_E^*)$.

\section*{Scope of Certainty}
Tasks 1--4, the constants $K_B,K_E$, the piecewise CDFs, the split structure, the saturation term, and term types (T1)--(T3) are \emph{exact closed forms}, verified symbolically (differentiation of the primitives) and numerically. The cross term (T4) provably admits no elementary/incomplete-Beta closed form for general non-integer $p_B,p_E$; its exact reduction (Appell $F_1$ series; finite elementary form for integer $p$) is given above, which is the strongest closed-form statement that mathematically exists for this problem. The accompanying MATLAB script \texttt{sop\_wiretap.m} evaluates the SOP via the analytic CDFs with the split at $\gamma_E^{*}$ and cross-checks against the brute-force double integral.
