\section{Revision Note: What Was Corrected and Why It Matters}
%% ==================================================================
The previous version of this guide (and the first release of the code) used the SNR relation
$v = b^{2} P_t \gamma P_o$ with a \emph{single shared} constant $b$ that \emph{multiplied} the SNR.
The correct model is
\begin{equation}
\boxed{\;
v_B \;=\; \frac{\gamma_B\, P_o}{b_1^{2}}\; P_t,
\qquad\qquad
v_E \;=\; \frac{\gamma_E\, P_o}{b_2^{2}}\; P_t
\;}
\label{eq:snr}
\end{equation}
i.e.\ (i) the optical constants \emph{divide} rather than multiply, and (ii) each link has its
\emph{own} constant: $b_1$ for Bob, $b_2$ for Eve. In the code these are folded into per-link
constants
\begin{equation}
c_B := \frac{P_o}{b_1^{2}}, \qquad c_E := \frac{P_o}{b_2^{2}},
\qquad\text{so}\qquad v_B = c_B\,P_t\,\gamma_B,\quad v_E = c_E\,P_t\,\gamma_E .
\end{equation}

\paragraph{Design implication (sign flip of intuition).} Because $b^{2}$ now sits in the
\emph{denominator}, a \emph{larger} $b_2$ \emph{weakens} Eve's SNR. Hence the favourable secrecy
regime is $b_2 > b_1$ --- the \emph{opposite} of the intuition under the old (incorrect)
multiplicative relation. All quantities that depend on the SNR relation were updated accordingly:
the environment reward, the closed-form reward gradient, the water-filling baseline, and CAPG's
analytical gradient. The learning \emph{algorithms} (DDPG, SAC, PPO, ECDPO) are untouched: they
interact with the channel only through the reward signal, so only their documentation changed.

\paragraph{Verification of the correction.} The updated analytical derivative
(equation~\eqref{eq:drdp} below) was checked against central-difference numerical differentiation
of the new reward using deliberately \emph{asymmetric} constants ($b_1{=}1.3$, $b_2{=}0.9$,
$P_o{=}1.2$, chosen to expose any residual shared-constant bug): agreement to
$3.5\times 10^{-10}$, and the water-filling solver returns a sensible interior optimum.

\section{The Underlying MDP Formulation (Corrected)}
%% ==================================================================

\subsection{Channel and SNR Model}
The instantaneous received SNRs follow \eqref{eq:snr}: $P_t$ is the transmit power (the
\emph{control action}), $\gamma_B,\gamma_E$ are the instantaneous fading gains, and
$b_1, b_2, P_o$ are fixed optical constants (constructor arguments
\texttt{WiretapEnv(b1=..., b2=..., P\_o=...)}; defaults $b_1{=}b_2{=}P_o{=}1$).

\subsection{Markov Decision Process}
\textbf{State.} The pair of fading gains, normalised to $[0,1]^2$:
\begin{equation}
s_t = \big(\gamma_B^{(t)},\, \gamma_E^{(t)}\big).
\end{equation}
The gains evolve as a first-order autoregressive (AR(1)) process modelling channel coherence:
\begin{equation}
\gamma^{(t+1)} \;=\; \rho\,\gamma^{(t)} + (1-\rho)\, w_t,
\qquad w_t \sim \mathrm{Unif}(\gamma_{\min},\gamma_{\max}),\qquad \rho = 0.85,
\label{eq:ar1}
\end{equation}
clipped to $[\gamma_{\min},\gamma_{\max}]=[0.2,\,3.0]$. The correlation $\rho=0.85$ makes
consecutive channel states strongly dependent, as in a physical fading channel within its
coherence time.

\textbf{Action.} Transmit power on a continuous range:
\begin{equation}
a_t = P_t \in [P_{\min}, P_{\max}] = [0.1,\; 5.0].
\end{equation}
The neural actor outputs a sigmoid value $a^{01}\in(0,1)$, affinely mapped to $[P_{\min},P_{\max}]$.

\textbf{Reward.} Instantaneous secrecy rate under the base-2 rate law $\log_2(1+Av)$ with the
corrected SNRs \eqref{eq:snr}, minus a Lagrangian penalty enforcing the average power budget
$\bar P = 2.0$:
\begin{equation}
\boxed{\;
r_t \;=\; \Big[\log_2\!\big(1 + A_B\, c_B\, P_t\, \gamma_B\big)
\;-\; \log_2\!\big(1 + A_E\, c_E\, P_t\, \gamma_E\big)\Big]^{+}
\;-\; \lambda\,\big(P_t - \bar P\big)
\;}
\label{eq:reward}
\end{equation}
with $[x]^+=\max(x,0)$, $A_B = 1.5 > A_E = 0.5$, and $\lambda = 0.3$.

\textbf{Objective.} Each agent maximises the discounted return
\begin{equation}
J(\pi) = \E_\pi\!\Big[\sum_{t=0}^{\infty} \gamma^{t}\, r_t\Big],\qquad \gamma = 0.99,
\end{equation}
equivalent up to scaling to maximising the long-run \emph{average} secrecy reward, i.e.\ the
adaptive-power ASC.

\subsection{The Analytical Baseline (Water-Filling, Corrected)}
The reward \eqref{eq:reward} is concave in $P_t$ for every fixed state, so the single-step optimum
has a closed form. The exact derivative under the corrected relation is
\begin{equation}
\frac{\partial r}{\partial P}
=\frac{1}{\ln 2}\left[
\frac{A_B\,c_B\,\gamma_B}{1 + A_B\,c_B\,P\,\gamma_B}
-\frac{A_E\,c_E\,\gamma_E}{1 + A_E\,c_E\,P\,\gamma_E}
\right] - \lambda,
\qquad c_B=\frac{P_o}{b_1^{2}},\;\; c_E=\frac{P_o}{b_2^{2}},
\label{eq:drdp}
\end{equation}
and the per-state optimal power $P^\ast(\gamma_B,\gamma_E)$ is obtained by bisection on
\eqref{eq:drdp} (\texttt{waterfill\_P} in \texttt{wiretap\_env.py}), with $P^\ast = P_{\min}$
whenever $A_B c_B\gamma_B \le A_E c_E\gamma_E$ (no secrecy possible at any power --- note the
per-link constants now enter this feasibility condition too). This baseline is the dashed grey
line in every learning-curve figure.

\section{Theory Behind Each of the Five Agents}
%% ==================================================================

\subsection{DDPG --- Deterministic Policy Gradient}
Actor $\mu_\theta(s)\!\to\!P$, critic $Q_\phi(s,a)$. The critic minimises TD error against
slowly-moving target networks,
\begin{equation}
\mathcal{L}(\phi)=\E\big[(Q_\phi(s,a) - y)^2\big],\qquad
y = r + \gamma\, Q_{\phi'}\big(s',\, \mu_{\theta'}(s')\big),
\end{equation}
and the actor ascends the deterministic policy gradient:
\begin{equation}
\nabla_\theta J = \E_s\Big[\nabla_a Q_\phi(s,a)\big|_{a=\mu_\theta(s)}\; \nabla_\theta\, \mu_\theta(s)\Big].
\end{equation}
Exploration: additive Gaussian noise on the sigmoid output (with linear decay in the 150k-step
configuration).

\subsection{SAC --- Maximum-Entropy RL}
Stochastic squashed-Gaussian actor; entropy-augmented objective
\begin{equation}
J_{\mathrm{SAC}} = \E\Big[\sum_t r_t + \alpha\, \mathcal H\big(\pi(\cdot\,|\,s_t)\big)\Big],
\end{equation}
twin critics with min-target,
\begin{equation}
y = r + \gamma\Big(\min_{i=1,2} Q_{\phi_i'}(s',a') - \alpha \log\pi(a'|s')\Big),\quad a'\sim\pi(\cdot|s'),
\end{equation}
temperature $\alpha$ auto-tuned toward target entropy $\bar{\mathcal H}=-1$.

\subsection{PPO --- Clipped Trust Region (On-Policy)}
Gaussian policy, GAE advantages ($\lambda_{\mathrm{GAE}}{=}0.95$), clipped surrogate
\begin{equation}
\mathcal{L}^{\mathrm{CLIP}}(\theta) = \E\Big[\min\big(\rho_t(\theta)\hat A_t,\;
\mathrm{clip}(\rho_t(\theta),\,1{-}\epsilon,\,1{+}\epsilon)\,\hat A_t\big)\Big],
\qquad \epsilon=0.2,
\end{equation}
optimised for $K{=}8$ epochs per 256-step rollout.

\subsection{ECDPO --- The Hybrid}
Deterministic sigmoid actor (DDPG) $+$ twin-Q min-target critics (SAC/TD3) $+$ Bernoulli-entropy
bonus on the $(0,1)$ action output:
\begin{equation}
\mathcal{L}_{\mathrm{actor}} = -\min_{i} Q_{\phi_i}\big(s,\mu_\theta(s)\big)
\;-\; c_{\mathrm{ent}}\Big[-a\log a-(1{-}a)\log(1{-}a)\Big],\qquad c_{\mathrm{ent}}=0.05.
\end{equation}

\subsection{CAPG --- Concavity-Aware Policy Gradient (Novel, Corrected)}
The key property: the reward gradient with respect to the action is available \emph{in closed
form} --- equation~\eqref{eq:drdp}, now with the per-link constants $c_B, c_E$. CAPG blends
\begin{equation}
\boxed{\;
\nabla_\theta J \;=\; (1-\beta)\underbrace{\nabla_\theta\, Q_\phi\big(s,\mu_\theta(s)\big)}_{\text{critic: long-horizon, learned}}
\;+\; \beta\, \underbrace{\frac{\partial r}{\partial P}\Big|_{P=\mu_\theta(s)} \nabla_\theta\,\mu_\theta(s)}_{\text{analytical: single-step, exact}}
\;}
\end{equation}
with a \emph{learnable} $\beta=\sigma(\log\beta)$. Under the corrected formula, $\beta$ again rose
rapidly (to $0.84$ within only $1{,}500$ steps), i.e.\ the agent learned to trust the physics.
Because the analytical term is exact and noise-free, CAPG remains the fastest-converging method;
its critic supplies the multi-step lookahead over the correlated AR(1) channel that greedy
water-filling lacks, which is why it can exceed the water-filling baseline.

corrected-model results. To regenerate: \texttt{train(steps=150000, eval\_every=5000)} in
\texttt{ddpg.py} ($\approx$9--10 min on a laptop CPU).

\section{Why the Learning Curves Fluctuate}
%% ==================================================================
Four distinct noise sources act simultaneously; the dominant one is \emph{measurement}, not the
agents. (Unchanged by the formula correction --- these are properties of the evaluation protocol
and of TD learning.)

\subsection{Source 1 (Dominant): Each Plotted Point Is a Monte-Carlo Estimate}
Every point is the average reward over $5$--$20$ episodes $\times$ $200$ steps of a \emph{random}
fading process. The per-state reward has high intrinsic variance: in favourable states the
secrecy term is large; whenever $A_B c_B\gamma_B \le A_E c_E\gamma_E$ the $[\cdot]^+$ clips it to
zero. With correlation $\rho = 0.85$, consecutive samples are strongly dependent, so for a
1000-sample evaluation the \emph{effective} number of independent samples is only
\begin{equation}
N_{\mathrm{eff}} \;\approx\; N\cdot \frac{1-\rho}{1+\rho}
\;=\; 1000 \times \frac{0.15}{1.85} \;\approx\; 81,
\end{equation}
giving a standard error of ${\sim}\sigma/9 \approx \pm 0.01$--$0.02$ per point --- exactly the
observed wiggle. \textbf{Decisive check:} the deterministic, per-step-optimal water-filling policy
shows the \emph{same} wiggle when re-evaluated with fresh seeds. The fluctuation is measurement
noise, not policy instability.

\subsection{Source 2: The Policy Changes Between Checkpoints}
Gradient updates occur every environment step. Near the optimum this costs little: the reward is
concave and \emph{flat} in $P$ around $P^\ast$, so small action errors produce tiny reward
differences --- fluctuations stay bounded.

\subsection{Source 3: Bootstrapped, Moving Targets}
The critic learns from targets built with its own (target-network) estimates,
$y = r+\gamma\,Q'(s',\cdot)$; the actor climbs a critic that is itself moving. Function
approximation $+$ bootstrapping $+$ off-policy data (the ``deadly triad'') oscillates \emph{around}
the fixed point rather than converging monotonically --- a theoretical property, not a bug.

\subsection{Source 4: Stochastic Minibatch Gradients}
Each update samples 128 random transitions from the replay buffer; every gradient step is a noisy
estimate. SAC additionally injects entropy-driven action noise into its own training data ---
hence the noisiest and slowest-rising early curve.

\section{Why the Reward Can Never Be a Constant Flat Line}
%% ==================================================================
Even the \emph{perfectly converged optimal policy} does not produce constant reward. What is
constant at convergence is the \emph{expectation}
\begin{equation}
\rho^\ast \;=\; \lim_{T\to\infty}\frac1T\sum_{t=1}^{T} r_t \qquad\text{(a single number)},
\end{equation}
but any \emph{finite-window} average is a random variable, because
$r_t = r\big(s_t,\pi^\ast(s_t)\big)$ depends on the random channel state $s_t$ --- and $s_t$
never stops fluctuating; that is the physics of fading. In a good state the optimal policy earns
${\sim}2$ bits of secrecy; in a state with $A_B c_B\gamma_B \le A_E c_E\gamma_E$ it earns zero
secrecy plus only the power-saving term. A flat reward would require a \emph{static} channel ---
in which case a single water-filling computation suffices and no DRL is needed. The fluctuation
is the signature of the problem being nontrivial.

\paragraph{Correct convergence criterion.} Not ``flat curve,'' but: \textbf{curve oscillating
within a fixed band around the water-filling line}.

\section{How to Obtain Smoother Curves (for Publication)}
%% ==================================================================
In priority order:
\begin{enumerate}[nosep]
\item \textbf{More evaluation episodes:} raise \texttt{n\_ep} from 5 to 30--50 (standard error
drops by ${\sim}\sqrt{10}$).
\item \textbf{Common random numbers:} fix the evaluation seed so every checkpoint sees the
\emph{identical} fading sequence (removes Source~1; already implemented in the 150k
\texttt{ddpg.py}).
\item \textbf{Exponential moving average} of the raw points (plot both).
\item \textbf{Anneal exploration noise and learning rate} late in training (shrinks Sources~2, 4;
noise decay already implemented in \texttt{ddpg.py}).
\item \textbf{Multiple seeds:} $\ge 5$ seeds, plot mean $\pm$ std band --- the standard the DRL
literature expects.
\end{enumerate}
