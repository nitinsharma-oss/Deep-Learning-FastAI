% =====================================================================
%  MACRO :  \StarGlow{x}{y}{radius}{coreradius}
%  A tiny glowing star = soft halo + bright core.
% =====================================================================
\newcommand{\StarGlow}[4]{%
  \begin{scope}[shift={(#1,#2)}]
    \fill[Gold,path fading=radial glow,opacity=0.35] (0,0) circle (#3);
    \fill[LightGold,opacity=0.9] (0,0) circle (#4);
  \end{scope}%
}

% =====================================================================
%  MACRO :  \Hexagon{x}{y}{size}
%  Single pointy-top hexagon outline (for the science lattice).
% =====================================================================
\newcommand{\Hexagon}[3]{%
  \draw[goldhair]
    ($(#1,#2)+(90:#3)$) -- ($(#1,#2)+(150:#3)$) --
    ($(#1,#2)+(210:#3)$) -- ($(#1,#2)+(270:#3)$) --
    ($(#1,#2)+(330:#3)$) -- ($(#1,#2)+(30:#3)$) -- cycle;%
}

% =====================================================================
%  MACRO :  \CornerOrnament
%  Decorative corner flourish drawn in the lower-left origin frame.
%  Placement/rotation handled by the caller via a scope.
% =====================================================================
\newcommand{\CornerOrnament}{%
  % nested right-angle brackets
  \draw[ornament] (0,2.6) -- (0,0) -- (2.6,0);
  \draw[ornamentthin] (0.28,2.2) -- (0.28,0.28) -- (2.2,0.28);
  % sweeping quarter arc
  \draw[ornamentthin] (1.9,0.28) arc (0:90:1.62);
  % little diamond accents along the arc
  \foreach \a in {18,45,72}{
    \node[Gold,scale=0.6] at ($(0.28,0.28)+(\a:1.62)$) {$\blacklozenge$};
  }
  % corner star burst
  \fill[Gold] (0.28,0.28) circle (1.6pt);
  \foreach \a in {0,45,...,315}{
    \draw[ornamentthin] (0.28,0.28) -- ++(\a:0.42);
  }
}

% =====================================================================
%  MACRO :  \GoldDivider{width}
%  Elegant centred rule: line — diamond — line, with end dots.
% =====================================================================
\newcommand{\GoldDivider}[1]{%
  \begin{tikzpicture}[baseline]
    \def\w{#1}
    \draw[Gold,line width=0.8pt] (-\w,0) -- (-0.32,0);
    \draw[Gold,line width=0.8pt] (0.32,0)  -- (\w,0);
    \node[Gold] at (0,0) {$\blacklozenge$};
    \fill[Gold] (-\w,0) circle (0.9pt);
    \fill[Gold] (\w,0)  circle (0.9pt);
  \end{tikzpicture}%
}

% =====================================================================
%  MACRO :  \Astrolabe{scale}
%  Astronomy icon: astrolabe / sunburst — concentric rings, radial
%  ticks, a compass rose and a central star.
% =====================================================================
\newcommand{\Astrolabe}[1]{%
  \begin{tikzpicture}[scale=#1]
    % outer glow halo
    \fill[Gold,path fading=radial glow,opacity=0.25] (0,0) circle (1.55);
    % concentric rings
    \draw[Gold,line width=1.1pt] (0,0) circle (1.20);
    \draw[Gold,line width=0.5pt] (0,0) circle (1.02);
    \draw[Gold,line width=0.5pt,opacity=0.7] (0,0) circle (0.72);
    % graduated tick ring
    \foreach \a in {0,15,...,345}{
      \draw[Gold,line width=0.5pt] (\a:1.02) -- (\a:1.20);
    }
    \foreach \a in {0,90,180,270}{
      \draw[Gold,line width=1.0pt] (\a:0.95) -- (\a:1.20);
    }
    % compass rose points
    \foreach \a in {0,90,180,270}{
      \fill[Gold,opacity=0.9] (\a:0.0) -- (\a:0.62) -- (\a+18:0.14) -- cycle;
      \fill[Gold,opacity=0.9] (\a:0.0) -- (\a:0.62) -- (\a-18:0.14) -- cycle;
    }
    \foreach \a in {45,135,225,315}{
      \fill[Gold,opacity=0.45] (\a:0.0) -- (\a:0.42) -- (\a+16:0.10) -- cycle;
      \fill[Gold,opacity=0.45] (\a:0.0) -- (\a:0.42) -- (\a-16:0.10) -- cycle;
    }
    % central boss + a tiny four-point star
    \fill[PrimaryNavy] (0,0) circle (0.16);
    \fill[Gold] (0,0) circle (0.10);
    \node[Gold,scale=0.9] at (0,0) {\textbf{$\star$}};
    % small orbiting star
    \StarGlow{0.85}{0.85}{0.22}{0.03}
  \end{tikzpicture}%
}

  \IfFileExists{images/#1}{%
    \includegraphics[height=2.6cm,keepaspectratio]{#1}%
  }{%
    % --- elegant placeholder -----------------------------------------
    \begin{tikzpicture}
      \node[
        minimum width=2.6cm, minimum height=2.6cm,
        rounded corners=4pt,
        draw=Gold, line width=0.8pt,
        top color=PageBG, bottom color=LightGold!30,
        drop shadow={shadow xshift=0.6pt,shadow yshift=-0.6pt,opacity=0.25}
      ] (ph) {};
      \draw[Gold,line width=0.5pt] (ph.north west)++(3pt,-3pt)
        rectangle ($(ph.south east)+(-3pt,3pt)$);
      \node[align=center,text=RoyalBlue,font=\titlefont\small]
        at (ph.center) {\itshape logo\\[-1pt]\scriptsize\color{Gold}#1};
    \end{tikzpicture}%
  }%
}

% =====================================================================
%  MACRO :  \TitleBox
%  Premium 3D embossed hardcover-style title plate.
%    Layered: blur shadow → ambient glow → navy gradient body →
%    metallic gold double border → inner light-gold highlight →
%    glossy top reflection → bevelled edges → embossed gold text.
% =====================================================================
\newcommand{\TitleBox}{%
\begin{tikzpicture}
  \def\bw{12.4}   % box width
  \def\bh{4.6}    % box height
  \def\rc{10pt}   % corner radius

  % (a) ambient outer glow
  \fill[Gold,path fading=radial glow,opacity=0.22]
    (0,0) ellipse ({\bw/1.7} and {\bh/1.4});

  % (b) drop / blur shadow beneath the plate
  \node[
    minimum width=\bw cm, minimum height=\bh cm,
    rounded corners=\rc, fill=DeepNavy,
    blur shadow={shadow blur steps=9, shadow xshift=0pt,
                 shadow yshift=-5pt, shadow blur radius=9pt, opacity=0.55}
  ] (shadowlayer) {};

  % (c) main body — navy vertical gradient
  \shade[rounded corners=\rc,
         top color=RoyalBlue, bottom color=DeepNavy]
    (-\bw/2,-\bh/2) rectangle (\bw/2,\bh/2);

  % (d) metallic gold double border
  \draw[Gold,line width=2.4pt,rounded corners=\rc]
    (-\bw/2,-\bh/2) rectangle (\bw/2,\bh/2);
  \draw[LightGold,line width=0.8pt,rounded corners=\rc-1pt]
    ($(-\bw/2,-\bh/2)+(3pt,3pt)$) rectangle ($(\bw/2,\bh/2)+(-3pt,-3pt)$);

  % (e) inner light-gold highlight frame
  \draw[LightGold,line width=0.5pt,opacity=0.55,rounded corners=\rc-3pt]
    ($(-\bw/2,-\bh/2)+(9pt,9pt)$) rectangle ($(\bw/2,\bh/2)+(-9pt,-9pt)$);

  % (f) frosted-glass / glossy top reflection (bevel highlight)
  \begin{scope}
    \clip[rounded corners=\rc]
      (-\bw/2,-\bh/2) rectangle (\bw/2,\bh/2);
    \fill[white,opacity=0.10]
      (-\bw/2,\bh/2) -- (\bw/2,\bh/2) -- (\bw/2,0.55) --
      (-\bw/2,1.35) -- cycle;                        % raked gloss band
    \fill[white,path fading=radial glow,opacity=0.06]
      (0,\bh/2) ellipse ({\bw/2} and 1.2);           % soft sheen
    % faint bottom inner shadow for depth
    \fill[DeepNavy,opacity=0.35]
      (-\bw/2,-\bh/2) rectangle (\bw/2,-\bh/2+0.35);
  \end{scope}

  % (g) embossed gold title text -------------------------------------
  % subtle dark shadow copy first (gives the raised / embossed feel)
  \node[align=center,opacity=0.45] at (0.4pt,0.85-0.03)
    {\titlefont\bfseries\color{DeepNavy}\fontsize{22}{26}\selectfont
       Gamma-Ray Emission Model};
  \node[align=center] at (0,0.85)
    {\titlefont\bfseries\color{LightGold}\fontsize{22}{26}\selectfont
       Gamma-Ray Emission Model};
  \node[align=center,text=LightGold] at (0,0.05)
    {\titlefont\itshape\fontsize{13}{15}\selectfont for};
  \node[align=center] at (0,-0.78)
    {\titlefont\bfseries\color{Gold}\fontsize{18}{22}\selectfont
       Astrophysical Radiation Analysis};

  % (h) acronym plate
  \node[align=center,text=white] at (0,-1.62)
    {\titlefont\scshape\fontsize{13}{15}\selectfont (GAMERA)};
\end{tikzpicture}%
}
