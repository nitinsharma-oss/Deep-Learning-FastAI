%------------------------------------------------------------------------------
%  Corner ornament: an arc flourish + diamond + accent strokes.
%  #1 = anchor coordinate, #2 = rotation (degrees)
%------------------------------------------------------------------------------
\newcommand{\cornerornament}[2]{%
  \begin{scope}[shift={#1},rotate=#2]
    % sweeping quarter arcs
    \draw[gold,line width=1.3pt] (0,0.95) arc (90:0:0.95);
    \draw[gold,line width=0.6pt] (0,0.72) arc (90:0:0.72);
    % little diamond where the arcs meet the border
    \fill[gold] (0.67,0.67) circle (0.055);
    \fill[navy] (0.90,0.10) -- (1.05,0.25) -- (0.90,0.40) -- (0.75,0.25) -- cycle;
    \fill[gold] (0.90,0.16) -- (0.99,0.25) -- (0.90,0.34) -- (0.81,0.25) -- cycle;
    % short accent tails along the frame
    \draw[gold,line width=0.9pt] (0,0)--(0,0.45);
    \draw[gold,line width=0.9pt] (0,0)--(0.45,0);
    \fill[gold] (0,0.55) circle (0.04);
    \fill[gold] (0.55,0) circle (0.04);
  \end{scope}}

%------------------------------------------------------------------------------
%  Logo loader: looks for the file in images/ first, then in the project root,
%  and only falls back to a neutral framed placeholder if neither exists — so
%  the file always compiles. Pass the bare file name, e.g. \logoslot{logo1.png}.
%  (Logos are never *drawn*.)
%------------------------------------------------------------------------------
\newcommand{\logoslot}[1]{%
  \IfFileExists{images/#1}%
    {\includegraphics[height=2.35cm]{images/#1}}%
    {\IfFileExists{#1}%
      {\includegraphics[height=2.35cm]{#1}}%
      {\begin{tikzpicture}
         \node[draw=navy,line width=1pt,rounded corners=3pt,
               minimum width=2.35cm,minimum height=2.35cm,align=center,
               text=navy,font=\small]{logo\\image};
       \end{tikzpicture}}}}
