\title{Simulation Results: DRL-Based OIRS Orientation Optimization and Outage
Performance of OIRS-Aided VLC Systems under Random UE Orientation}
\date{\today}

\begin{document}
\maketitle

\noindent
All results are generated by \texttt{train\_all.py}: five DRL agents (DDPG,
SAC, PPO, CAPG, ECDPO) trained under identical conditions ($K{=}3$ users,
$M{=}1$, $N{=}4$, truncated-Laplace UE orientation with $b_\xi{=}0.036$,
$\xi\in[0,0.9574]$, 6{,}000 stochastic interactions each), benchmarked
against a common-random-number grid search, followed by the min--max AoI
fairness study and the analytical/Monte-Carlo outage validation.

\begin{figure}[!ht]
\centering
\includegraphics[width=0.92\textwidth]{fig1_convergence.pdf}
\caption{\textbf{Convergence.} Moving average of the sampled mean ADR per
environment interaction for the five DRL algorithms versus the grid-search
reference (37.12~Mbit/s, dashed). The off-policy agents (SAC, ECDPO, DDPG)
approach the expected-ADR optimum; the on-policy agents (PPO, CAPG) remain
sample-starved at this budget under the noisy stochastic reward.}
\end{figure}

\begin{figure}[!ht]
\centering
\includegraphics[width=0.80\textwidth]{fig2_final_adr.pdf}
\caption{\textbf{Final performance.} Expected mean ADR of each learned
orientation, evaluated with common random numbers over the UE-orientation
statistics. SAC (37.11), ECDPO (37.06), and DDPG (36.94) reach within
$0.5\,\%$ of the grid reference (dashed).}
\end{figure}

\begin{figure}[!ht]
\centering
\includegraphics[width=0.95\textwidth]{fig3_fairness.pdf}
\caption{\textbf{Throughput/fairness trade-off.} Per-user expected ADR
(left) and average HARQ-IR Age of Information $\overline{\Delta}_{k,IR}$
(right) at the ADR-optimal orientation (orange) versus the AoI-optimal
orientation learned with the min--max AoI reward on the wide-beam
configuration ($\kappa{=}6$, $\bar{\gamma}{=}3{\times}10^{15}$; green).
Pure rate maximization abandons UE~1 entirely (0~Mbit/s, worst-case AoI 4.5
rounds); the AoI-trained agent attains the theoretical minimum AoI of 1.5
rounds for \emph{every} user, matching the min--max grid reference exactly.}
\end{figure}

\begin{figure}[!ht]
\centering
\includegraphics[width=0.92\textwidth]{fig4_outage.pdf}
\caption{\textbf{Outage validation and zero-outage cliff.} Outage probability
versus average transmit SNR at the learned orientation (best-served user,
$R{=}2$, $\xi_{\min}{=}0.2$): closed-form Gamma/Meijer-G analysis (lines)
against Monte Carlo simulation (markers). The truncated-Laplace orientation
model reaches $P_{out}{=}10^{-6}$ roughly 8~dB before the bounded uniform
model, and both drop \emph{exactly} to zero at the analytical zero-outage
cliff $\bar{\gamma}_{\mathrm{cliff}}\approx 146$~dB (dotted), whereas the
conventional uniform assumption with $\xi_{\min}\!\to\!0$ exhibits no cliff
and its outage tail never terminates.}
\end{figure}
